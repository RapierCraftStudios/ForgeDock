Edge functions

Server-side logic executed at the edge. Standard Web APIs. No cold starts. Deployed alongside your frontend automatically.

app/api/analytics/route.ts
export const runtime = 'edge';

export async function POST(req: Request) {
  const { event, properties } = await req.json();
  const geo = req.geo;

  // Process at the nearest edge region
  await analytics.track({
    event,
    properties,
    context: {
      region: process.env.SLIPSTREAM_REGION,
      city: geo?.city,
      country: geo?.country,
      timestamp: Date.now(),
    },
  });

  return Response.json({
    status: 'ok',
    region: process.env.SLIPSTREAM_REGION,
    latency: `${Date.now() - start}ms`,
  });
}
