@voltage.endpoint(
    gpu="T4",
    concurrency=50,
)
def generate(prompt: str):
    from vllm import LLM

    llm = LLM(
        model="mistralai/Mistral-7B"
    )
    output = llm.generate(prompt)
    return {
        "text": output[0].text
    }

@voltage.schedule(
    cron="0 3 * * *",
    memory=16384,
    cpu=4,
)
def nightly_etl():
    import polars as pl

    df = pl.read_parquet("s3://...")
    result = (
        df.filter(pl.col("date") ==
                  yesterday())
          .group_by("region")
          .agg(pl.col("revenue").sum())
    )
    result.write_delta("s3://...")
