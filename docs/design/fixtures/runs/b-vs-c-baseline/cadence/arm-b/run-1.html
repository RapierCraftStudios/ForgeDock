import { Cadence } from '@cadence/sdk';

const client = new Cadence({
  apiKey: process.env.CADENCE_API_KEY,
});

await client.email.send({
  to:       'maya@example.com',
  from:     'hello@yourapp.com',
  template: 'welcome-v2',
  data: {
    name:      'Maya',
    plan:      'pro',
    loginUrl:  'https://app.yourapp.com/login',
  },
});
