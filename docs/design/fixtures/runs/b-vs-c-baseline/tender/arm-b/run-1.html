create-payment.js
// Install: npm install @tender/sdk
import Tender from '@tender/sdk';
const tender = new Tender(process.env.TENDER_SECRET_KEY);
const payment = await tender.payments.create({
amount: 4999, // $49.99
currency: 'usd',
payment_method: 'pm_card_visa',
capture_method: 'automatic',
metadata: { order_id: 'ord_8821' }
});
// → { id: 'pay_3Kx92m...', status: 'succeeded' }

create-subscription.py Python 
import tender

client = tender.Client("sk_live_...")

subscription = client.subscriptions.create(
    customer="cus_7Gh3pa",
    items=[{
        "price": "price_growth_monthly",
        "quantity": 5     # seats
    }],
    trial_period_days=14,
    proration_behavior="create_prorations"
)
