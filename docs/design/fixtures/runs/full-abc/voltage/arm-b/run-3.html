A10G · 24GB VRAM
import voltage
from vllm import LLM, SamplingParams
from typing import Iterator


@voltage.endpoint(
    gpu="A10G",
    min_containers=1,         # keep one warm for low-latency
    max_containers=20,        # burst under traffic
    scaledown_window=300,    # 5 min grace period
)
class LlamaServer:
    def __init__(self):
        # Runs once — model loaded into warm container
        self.llm = LLM(
            model="meta-llama/Llama-3.1-8B-Instruct",
            tensor_parallel_size=1,
            dtype="bfloat16",
        )
        self.params = SamplingParams(temperature=0.7, max_tokens=512)

    @voltage.web_endpoint(method="POST", streaming=True)
    def generate(self, prompt: str) -> Iterator[str]:
        for output in self.llm.generate_streaming([prompt], self.params):
            yield output.outputs[0].text


# Deploy — returns a live HTTPS endpoint
# $ voltage deploy inference_server.py::LlamaServer
# → https://api.voltage.run/org/llama-server/generate

train_diffusion.py 
import voltage
from diffusers import DDPMPipeline


@voltage.function(
    gpu="A100-80GB",
    memory="96GB",
    cpu=8,
    timeout=14400,  # 4 hours
)
def train_diffusion_model(
    config_path: str,
    output_path: str,
) -> dict:
    pipeline = DDPMPipeline.from_pretrained(config_path)
    pipeline.train(output_dir=output_path)
    return pipeline.metrics()


# $ python train_diffusion.py
metrics = train_diffusion_model.remote(
    config_path="./configs/sd-xl.yaml",
    output_path="s3://models/sd-xl-v2",
)
hyperparam_sweep.py 
import itertools, voltage


@voltage.function(gpu="T4", memory="16GB")
def train_run(lr: float, dropout: float, layers: int) -> float:
    model = build_model(layers, dropout)
    return fit(model, lr=lr).best_val_loss


# 180 hyperparameter combinations
grid = [
    {"lr": lr, "dropout": d, "layers": l}
    for lr, d, l in itertools.product(
        [1e-4, 3e-4, 1e-3],
        [0.0, 0.1, 0.2, 0.3],
        [4, 6, 8, 12, 16],
    )
]

# All 180 run in parallel on separate GPU containers
results = voltage.map(train_run, grid)
best = min(results, key=lambda r: r)
print(f"Best val loss: {best:.4f}")
# 180 runs · 47 minutes · $12.40
