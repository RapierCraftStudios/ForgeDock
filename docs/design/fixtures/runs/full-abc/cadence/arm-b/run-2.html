send-order-confirmation.ts 
import { Cadence } from '@cadence/sdk';

// Initialize once — reuse across your app
const cadence = new Cadence({ apiKey: process.env.CADENCE_API_KEY });

// In your order handler:
async function onOrderPlaced(order: Order) {
  const result = await cadence.email.send({
    to:        order.customer.email,
    from:      'orders@acmecorp.com',
    template: 'order-confirmed',
    data: {
      orderNumber: order.id,
      items:       order.lineItems,
      total:       order.total,
      trackingUrl: order.tracking,
    },
    // Safe to retry — won't duplicate-send
    idempotencyKey: `order-confirm-${order.id}`,
  });

  // { id: 'msg_01HX...', status: 'queued', estimatedDelivery: '...' }
  console.log('Email queued:', result.id);
}
